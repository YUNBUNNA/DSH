អាហារូបត្ថម្ភ និងការពារជំងឺតាមរយៈកីឡា

របៀបបញ្ចូលអាហារល្អ និងការហាត់ប្រាណ ដើម្បីកសាងសុខភាពរឹងមាំ និងការពារជំងឺរ៉ាំរ៉ៃ

លោក ចាន់ សុវណ្ណ

អ្នកជំនាញអាហារូបត្ថម្ភកីឡា • ថ្ងៃទី២៨ ខែមិថុនា ២០២៤

អាហារូបត្ថម្ភល្អ រួមផ្សំជាមួយការហាត់ប្រាណជាប្រចាំ គឺជាគន្លឹះសំខាន់បំផុតក្នុងការរក្សាសុខភាពរឹងមាំ និងការពារជំងឺជាច្រើនប្រភេទ។ ការយល់ដឹងអំពីអាហារត្រឹមត្រូវ និងការហាត់ប្រាណសមស្រប អាចជួយកាត់បន្ថយហានិភ័យជំងឺរ៉ាំរ៉ៃ និងធ្វើឱ្យជីវិតមានគុណភាពកាន់តែប្រសើរ។

តើកីឡាជួយការពារជំងឺអ្វីខ្លះ?

ជំងឺបេះដូង និងសរសៃឈាម

ការហាត់ប្រាណជាប្រចាំជួយពង្រឹងសាច់ដុំបេះដូង កាត់បន្ថយសម្ពាធឈាម និងកាត់បន្ថយកូឡេស្តេរ៉ុលអាក្រក់ក្នុងឈាម។ វាជួយឱ្យសរសៃឈាមបត់បែន និងកាត់បន្ថយហានិភ័យគាំងបេះដូង និងជំងឺសរសៃឈាមខួរក្បាល។

ជំងឺទឹកនោមផ្អែម ប្រភេទ២

សកម្មភាពរាងកាយជួយគ្រប់គ្រងកម្រិតស្ករក្នុងឈាម ដោយបង្កើនប្រសិទ្ធភាពអាំងស៊ុយលីន។ ការហាត់ប្រាណជាប្រចាំ រួមជាមួយអាហារត្រឹមត្រូវ អាចទប់ស្កាត់ ឬពន្យាពេលការកើតជំងឺទឹកនោមផ្អែម និងជួយគ្រប់គ្រងជំងឺបានប្រសើរជាងមុន។

ជំងឺមហារីក

ការហាត់ប្រាណជាប្រចាំជួយកាត់បន្ថយហានិភ័យជំងឺមហារីកមួយចំនួន ដូចជាមហារីកពោះវៀន មហារីកសុដន់ និងមហារីកសួត។ វាជួយរក្សាទម្ងន់ល្អ គ្រប់គ្រងអរម៉ូន និងពង្រឹងប្រព័ន្ធភាពសុំាររាងកាយ។

ជំងឺផ្លូវដង្ហើម

ការហាត់ប្រាណជួយពង្រឹងសួត និងសាច់ដុំដង្ហើម។ វាកាត់បន្ថយហានិភ័យជំងឺរ៉ាំរ៉ៃ ដូចជាហឺត និងជំងឺផ្លូវដង្ហើមរ៉ាំរ៉ៃ។ អ្នកដែលហាត់ប្រាណជាប្រចាំមានសមត្ថភាពសួតកាន់តែល្អ។

ជំងឺឆ្អឹង

ការហាត់ប្រាណដែលទម្ងន់រាងកាយទ្រទ្រង់ ដូចជាការរត់ ការលោត ឬការលើកទម្ងន់ ជួយបង្កើនដង់ស៊ីតេឆ្អឹង និងការពារជំងឺឆ្អឹងពុក។ វាជាការសំខាន់ជាពិសេសសម្រាប់មនុស្សវ័យចំណាស់។

អាហារូបត្ថម្ភសំខាន់សម្រាប់អ្នកហាត់ប្រាណ

1

ប្រូតេអ៊ីន (Protein)

ប្រូតេអ៊ីនចាំបាច់សម្រាប់ការជួសជុល និងកសាងសាច់ដុំ។ ប្រភពល្អដូចជា៖ សាច់មាន់ ត្រី ស៊ុត ដំឡូងបារាំង សណ្តែកសៀង និងទឹកដោះគោ។ អ្នកហាត់ប្រាណគួរទទួលទាន ១.២-២ ក្រាមក្នុង១គីឡូក្រាមទម្ងន់រាងកាយ។

2

កាបូអ៊ីដ្រាត (Carbohydrates)

កាបូអ៊ីដ្រាតជាប្រភពថាមពលសំខាន់បំផុតសម្រាប់ការហាត់ប្រាណ។ ជ្រើសរើសកាបូអ៊ីដ្រាតស្មុគ្រស្មាញ ដូចជា៖ បាយសំប៉ែត នំប័ុងសំប៉ែត ដំឡូងបាត៉ាត និងបង្គោល។ ជៀសវាងស្ករស និងអាហារកែច្នៃខ្លាំង។

3

ខ្លាញ់ល្អ (Healthy Fats)

ខ្លាញ់ល្អចាំបាច់សម្រាប់ការស្រូបយកវីតាមីន និងការផលិតអរម៉ូន។ ប្រភពល្អដូចជា៖ ផ្លែបឺរ ប្រេងអូលីវ គ្រាប់ធញ្ញជាតិ ត្រីខ្លាញ់ និងគ្រាប់ឈីយ៉ា។ ជៀសវាងខ្លាញ់ត្រង់ (Trans fats)។

4

វីតាមីន និងសារធាតុរ៉ែ

សារធាតុទាំងនេះចាំបាច់សម្រាប់ការងារជាច្រើនរបស់រាងកាយ។ ទទួលទានបន្លែ និងផ្លែឈើចម្រុះពណ៌ យ៉ាងតិច ៥ ក្រុមក្នុងមួយថ្ងៃ។ ពណ៌ផ្សេងៗផ្តល់សារធាតុផ្សេងៗគ្នា។

5

ទឹក

ទឹកជាសារធាតុសំខាន់បំផុត។ ផឹកទឹក ២-៣ លីត្រក្នុងមួយថ្ងៃ។ បន្ថែមបរិមាណនៅពេលហាត់ប្រាណ ឬពេលអាកាសធាតុក្តៅ។ សញ្ញាខ្សោយទឹកគឺ នៅយឺត លំហួរក្តៅ និងក្រហមក្បាល។

តារាងអាហារប្រចាំថ្ងៃ សម្រាប់អ្នកហាត់ប្រាណ

អាហារពេលព្រឹក (៦-៧ព្រឹក)

ជម្រើសទី១: បាយសំប៉ែត + ស៊ុត + ផ្លែឈើ + ទឹកដោះគោ

ជម្រើសទី២: នំប័ុងសំប៉ែត + បឺរ + ស៊ុត + ទឹកផ្លែឈើ

អាហារពេលព្រឹកចាំបាច់ដើម្បីផ្តល់ថាមពលសម្រាប់ពេញមួយថ្ងៃ

អាហារសម្រន់ពេលព្រឹក (៩-១០ព្រឹក)

ជម្រើសទី១: យ៉ាអួ + គ្រាប់ធញ្ញជាតិ

ជម្រើសទី២: ផ្លែឈើ + ទឹកដោះគោសូយ៉ា

រក្សាថាមពលរវាងអាហារធំ

អាហារថ្ងៃត្រង់ (១២-១ថ្ងៃត្រង់)

ជម្រើសទី១: បាយសំប៉ែត + សាច់ក្រលេញ + បន្លែ + ផ្លែឈើ

ជម្រើសទី២: មី + ត្រី + បន្លែ + ទឹកផ្លែឈើ

អាហារធំបំផុតនៃថ្ងៃ ត្រូវមានប្រូតេអ៊ីន និងបន្លែច្រើន

អាហារសម្រន់រសៀល (៣-៤រសៀល)

មុនហាត់ប្រាណ: ចេក + គ្រាប់ធញ្ញជាតិ + ទឹក

ក្រោយហាត់ប្រាណ: ស្មូធី ប្រូតេអ៊ីន + ផ្លែឈើ

ទទួលទានអាហារសម្រន់ មុនហាត់ប្រាណ ៣០-៦០នាទី

អាហារពេលល្ងាច (៦-៧ល្ងាច)

ជម្រើសទី១: បាយ + ត្រី + បន្លែឆ្អិន + សម្ល

ជម្រើសទី២: សាច់មាន់ + ដំឡូងបាត៉ាត + សាឡាត់

គួរតែស្រាល និងងាយស្រួលរំលាយ ជៀសវាងអាហារខ្លាញ់ច្រើន

អាហារដែលគួរជៀសវាង

អាហារឆាប់រហ័ស (Fast Food)

មានខ្លាញ់ អំបិល និងកាឡូរីខ្ពស់ ប៉ុន្តែមានសារធាតុចិញ្ចឹមតិច។ អាចបណ្តាលឱ្យធាត់ លើសទម្ងន់ និងជំងឺបេះដូង។

ភេសជ្ជៈផ្អែម និងជាតិស្ករខ្ពស់

ផ្តល់កាឡូរីទទេ គ្មានសារធាតុចិញ្ចឹម។ បង្កើនហានិភ័យជំងឺទឹកនោមផ្អែម និងធ្វើឱ្យធាត់។ ជំនួសដោយទឹក ឬទឹកផ្លែឈើធម្មជាតិ។

អាហារកែច្នៃខ្លាំង

មានសារធាតុរក្សា ពណ៌ និងជាតិផ្អែមសិប្បនិម្មិត។ ខ្វះសារធាតុចិញ្ចឹម និងអាចប៉ះពាល់ដល់សុខភាពរយៈពេលវែង។

រារាំងការស្រូបយកសារធាតុចិញ្ចឹម ធ្វើឱ្យខ្សោយទឹក និងប៉ះពាល់ដល់ការស្ទុះសាច់ដុំ។ គួរជៀសវាង ឬកំណត់បរិមាណ។

គន្លឹះសំខាន់សម្រាប់សុខភាពល្អ

ទទួលទានទៀងទាត់

ទទួលទានអាហារ ៥-៦ ដងក្នុងមួយថ្ងៃ ក្នុងបរិមាណតូច។ រក្សាពេលវេលាទទួលទានឱ្យទៀងទាត់។

ផឹកទឹកគ្រប់គ្រាន់

ផឹកទឹកឱ្យបានច្រើន មុន ក្នុងអំឡុង និងក្រោយការហាត់ប្រាណ។ សញ្ញាទឹកនោមស្រាលគឺសញ្ញាល្អ។

រៀបចំផែនការ

រៀបចំផែនការអាហារប្រចាំសប្តាហ៍។ នេះជួយសន្សំសំចៃពេលវេលា និងធានាឱ្យទទួលទានអាហារត្រឹមត្រូវ។

គេងលក់គ្រប់គ្រាន់

គេង ៧-៩ ម៉ោងក្នុងមួយយប់។ ការគេងលក់គ្រប់គ្រាន់ជួយស្ទុះសាច់ដុំ និងពង្រឹងប្រព័ន្ធភាពសុំារ។

ពេលណាត្រូវពិគ្រោះជាមួយអ្នកជំនាញ?

អ្នកគួរតែពិគ្រោះជាមួយអ្នកជំនាញអាហារូបត្ថម្ភ ឬគ្រូពេទ្យក្នុងករណី៖

  • ចង់ស្គមទម្ងន់ ឬបង្កើនទម្ងន់យ៉ាងច្រើន
  • មានអាឡែរហ្សីអាហារ ឬមិនអត់ទ្រាំអាហារមួយចំនួន
  • កំពុងហ្វឹកហាត់សម្រាប់ការប្រកួតកីឡា
  • ត្រូវការផែនការអាហារពិសេស
